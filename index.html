Sync & backup (Google)

Syncs the whole project (images included) through your own Google Drive, and keeps a readable text copy in your sheet.

One-time setup (about 3 minutes)
  1. Create a Google Sheet (sheets.new)
  2. Extensions → Apps Script, delete what's there, paste the code below
  3. Deploy → New deployment → type: Web app
  4. Execute as: Me • Who has access: Anyone
  5. Copy the web app URL, paste it above on every device, tick Auto-sync
  6. Already set up the older backup script? Paste this code over it, then Deploy → Manage deployments → pencil → Version: New version → Deploy (the URL stays the same)
function doGet(){
  var f=getFile_();
  var out=f?f.getBlob().getDataAsString():'{"empty":true}';
  return ContentService.createTextOutput(out)
    .setMimeType(ContentService.MimeType.JSON);
}
function doPost(e){
  var data=JSON.parse(e.postData.contents);
  var p=data.project||data;
  var body=JSON.stringify({project:p,
    savedAt:new Date().toISOString()});
  var f=getFile_();
  if(f){f.setContent(body);}
  else{DriveApp.createFile('mobius-storyboard-data.json',body);}
  try{updateSheets_(p);}catch(err){}
  return ContentService.createTextOutput('{"ok":true}')
    .setMimeType(ContentService.MimeType.JSON);
}
function getFile_(){
  var it=DriveApp.getFilesByName('mobius-storyboard-data.json');
  return it.hasNext()?it.next():null;
}
function updateSheets_(p){
  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var log=ss.getSheetByName('Backups')||ss.insertSheet('Backups');
  if(log.getLastRow()===0)
    log.appendRow(['Saved at','Threads','Scenes','Notes']);
  log.appendRow([new Date(),(p.threads||[]).length,
    (p.entries||[]).length,(p.notes||[]).length]);
  var sc=ss.getSheetByName('Scenes')||ss.insertSheet('Scenes');
  sc.clear();
  sc.appendRow(['Date','Thread','Character','Title',
    'Location','Archived','Text']);
  var tmap={};
  (p.threads||[]).forEach(function(t){tmap[t.id]=t;});
  (p.entries||[]).forEach(function(en){
    var t=tmap[en.threadId]||{};
    var txt=(en.html||'').replace(/<[^>]+>/g,' ')
      .replace(/\s+/g,' ').trim().slice(0,45000);
    sc.appendRow([en.date,t.name||'',t.character||'',
      en.title,en.location,en.archived?'yes':'',txt]);
  });
  var no=ss.getSheetByName('Notes')||ss.insertSheet('Notes');
  no.clear();
  no.appendRow(['Pinned to','Note','Archived']);
  (p.notes||[]).forEach(function(n){
    no.appendRow([n.type+(n.value?': '+n.value:''),
      n.text,n.archived?'yes':'']);
  });
}
